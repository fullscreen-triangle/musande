\begin{figure*}[!htbp]
    \centering
    \includegraphics[width=\textwidth]{validation/figures/panel_1.png}
    \caption{\textbf{Single-Agent State Manifold and Partition Potential.}
    (\textbf{A})~Synchronisation potential $V_{\rm sync}(R; K) = (K_c - K)R^2/2 + KR^4/4$ on the linear $y$-axis versus the internal Kuramoto order parameter $R\in[0,1]$ on the linear $x$-axis, for three coupling regimes: subcritical ($K=0.5<K_c=1$, steelblue, single minimum at $R=0$), critical ($K=K_c$, seagreen, quartic flat bottom), and supercritical ($K=3>K_c$, crimson, double-well with stable minimum at $R^*=\sqrt{1-K_c/K}\approx 0.816$). The pitchfork bifurcation at $K=K_c$ is the only genuine phase transition in the model. Empirical content of Theorem~\ref{thm:bifurcation} (Synchronisation bifurcation).
    (\textbf{B})~Variance-floor term $V_{\rm var}(\sigma^2; T, K) = k_B T \sigma^2 + k_B T/(K\sigma^2)$ of the partition potential (\Cref{def:potential}) on the linear $y$-axis versus phase variance $\sigma^2$ on the linear $x$-axis, for three coupling strengths $K\in\{1,2,4\}$. Each curve has a unique minimum at $\sigma^2_{\min}=\sqrt{1/K}$ (vertical dashed lines). Higher coupling $K$ produces a lower variance floor.
    (\textbf{C})~Variance floor scaling on log-log axes: $\sigma^2_{\min}$ versus $K$, six measurement points (steelblue), with theoretical line $\sigma^2_{\min}=k_BT/K$ (crimson). Slope is exactly $-1$ on log-log axes, confirming the analytical prediction. Experiment E05 reports relative error $1.11\times 10^{-16}$.
    (\textbf{D})~Three-dimensional surface of the partition potential $\Phi(R,\sigma^2)=V_{\rm sync}+V_{\rm var}+V_{\rm SF}+V_{\rm ent}$ over $(R,\sigma^2)\in[0.05,0.95]\times[0.1,1.5]$, $25\times 25$ grid, viridis colourmap, fixed $K=2$, $K_c=0.5$, $S$-entropy $\Svec=(0.5,0.5,0.5)$. The global minimum lies at $(R^*,\sigma^2_{\min})=(0.866,0.707)$ (the coherent regime); the surface rises sharply at low $\sigma^2$ (variance-floor barrier) and at $R=0$ (turbulent saddle).}
    \label{fig:panel1-state}
\end{figure*}


\begin{figure*}[!htbp]
    \centering
    \includegraphics[width=\textwidth]{validation/figures/panel_2.png}
    \caption{\textbf{Five Operational Regimes and Bifurcation.}
    (\textbf{A})~Bifurcation diagram: stable order parameter $R^*=\sqrt{1-K_c/K}$ on the linear $y$-axis versus coupling $K$ on the linear $x$-axis ($0$ to $5$), with $K_c=1.0$ (crimson dashed). For $K<K_c$, $R^*=0$ (incoherence); for $K>K_c$, $R^*$ rises continuously, asymptoting to $1$ as $K\to\infty$. The classical Kuramoto pitchfork.
    (\textbf{B})~The five operational bands on the $R$-axis: filled regions colour-coded (turbulent crimson, $R<0.3$; partial darkorange, $0.3\le R<0.5$; cascade gold, $0.5\le R<0.8$; coherent seagreen, $0.8\le R<0.95$; phase-locked steelblue, $R\ge 0.95$). Vertical dashed lines mark the four band cut-offs. These are classification labels on a continuous variable, \emph{not} transitions (\Cref{def:bands}, \Cref{prop:not-transitions}); the potential has no curvature sign-change at them.
    (\textbf{C})~Band widths under uniform sampling: bar chart of band occupancy across $1000$ uniform samples of $R\in[0,1]$. The bands occupy widths $0.3$, $0.2$, $0.3$, $0.15$, $0.05$; the bar heights are exactly proportional to the interval widths, illustrating that the cut-offs merely partition the interval.
    (\textbf{D})~Three-dimensional saddle landscape of $V_{\rm sync}(R,K)$ over $(R,K)\in[0,1]\times[0.1,4]$, $50\times 50$ grid, viridis colourmap. The bifurcation ridge at $K=K_c=1$ is clearly visible: below $K_c$ the surface has a single trough at $R=0$; above $K_c$ a double-well structure emerges with stable minima at $R^*$. Each regime corresponds to a distinct basin or saddle of $V_{\rm sync}$.}
    \label{fig:panel2-regimes}
\end{figure*}


\begin{figure*}[!htbp]
    \centering
    \includegraphics[width=\textwidth]{validation/figures/panel_3.png}
    \caption{\textbf{Ensemble Kuramoto and the Operational Bands at Ensemble Level.}
    (\textbf{A})~Ensemble Kuramoto order parameter $R_{\rm ens}$ on the linear $y$-axis versus uniform individual order parameter $R_i$ on the linear $x$-axis ($0$ to $1$), for an ensemble of $n=10$ agents with phases evenly distributed in $[0,\pi]$. The four band cut-offs are marked (crimson dashed). The relationship is monotonic but sub-linear because of partial phase cancellation. The bands are the same operational labels of \Cref{def:bands} read on $R_{\rm ens}$.
    (\textbf{B})~Synchronisation onset: $R_{\rm ens}$ vs inter-agent coupling $K$ for natural-frequency spread $\sigma_\omega=1$. Below the critical coupling $K_c$ (crimson dashed) $R_{\rm ens}=0$ (no synchronisation); above $K_c$, $R_{\rm ens}$ rises, crossing the phase-lock cut-off $0.95$ (steelblue dashed). The onset is the mean-field bifurcation of \Cref{thm:Kc} applied at the ensemble level (\Cref{thm:ens-sync}).
    (\textbf{C})~Phase distribution effect: bar chart comparing $R_{\rm ens}$ for two phase distributions across ensemble sizes $n\in\{2,3,4,5,6,8\}$. Crimson bars (uniform phase distribution): $R_{\rm ens}$ is small or zero, despite each agent having internal $R_i=1$. Seagreen bars (aligned phases): $R_{\rm ens}=1$. Demonstrates the hierarchical independence noted after \Cref{def:ens-R}: high $R_i$ does not imply high $R_{\rm ens}$.
    (\textbf{D})~Three-dimensional phase diagram: predicted $R_{\rm ens}$ over $(n_{\rm agents},\sigma_\omega)\in\{2,\dots,10\}\times[0.1,2.0]$ with fixed coupling $K=2$. The surface (viridis) shows that synchronisation strength decreases with frequency spread $\sigma_\omega$ (more diverse agents harder to synchronise) and is approximately independent of agent count when $K$ is fixed.}
    \label{fig:panel3-ensemble}
\end{figure*}


\begin{figure*}[!htbp]
    \centering
    \includegraphics[width=\textwidth]{validation/figures/panel_4.png}
    \caption{\textbf{Partition Extinction in the Categorical Model.}
    (\textbf{A})~Inter-agent partition lag $\tau_p$ on the linear $y$-axis versus a normalised control parameter $\theta/\theta_c$ on the linear $x$-axis ($0.5$ to $2.5$), with critical value $\theta_c=1$ (crimson dashed). For $\theta<\theta_c$ (distinguishable) $\tau_p>0$; at phase-locking ($\phi_i=\phi_j$, $\theta\ge\theta_c$) $\tau_p\equiv 0$. The seagreen-shaded region is the partition-extinct phase. The discontinuous drop is the signature of the binary distinguishability of \Cref{def:distinguishability}. Empirical content of \Cref{thm:extinction} (partition extinction is discontinuous); as \Cref{rem:correspondence-scope} states, the exact zero holds \emph{in the categorical model}, real systems retaining a small positive lag.
    (\textbf{B})~Coordination friction $\Theta_{\rm coord}=\frac1n\sum_{ij}\tau_p^{(ij)}G_{ij}$ on the linear $y$-axis versus the fraction of phase-locked pairs on the linear $x$-axis ($0$ to $1$). Friction falls to zero exactly when all pairs are phase-locked ($\phi_i=\phi_j$), the content of \Cref{prop:correspondence}. The step reflects the binary distinguishability of \Cref{def:distinguishability}: a pair is either distinguishable ($\tau_p>0$) or extinct ($\tau_p=0$), with no intermediate value. (This binary jump is a property of the pairwise lag, not a thermodynamic transition at any band cut-off; cf.\ \Cref{prop:not-transitions}.)
    (\textbf{C})~Pairwise categorical difference $d_{\rm cat}(\agent_i,\agent_j)$ on the linear $y$-axis versus phase difference $|\phi_i-\phi_j|$ on the linear $x$-axis ($0$ to $2$): monotone (steelblue), reaching zero exactly at $\phi_i=\phi_j$ (seagreen dashed), where the pair becomes categorically indistinguishable (\Cref{thm:extinction}, Step~2).
    (\textbf{D})~Three-dimensional comparison of composite floor surfaces over $(n_{\rm agents},f_{\rm per-agent})\in\{2,\dots,10\}\times[0.5,5]$. Lower surface (viridis): independent agents under parallel composition, $\Sfloor(\Compose)=\prod f_i/\Sigma^{n-1}$ (decreases with $n$). Upper surface (plasma): synchronised agents, $\Sfloor=\Sfloor(\agent_1)$ (constant in $n$). The two surfaces meet only at $n=1$; for $n\ge 2$ independent composition is strictly lower (more agents help) but synchronised composition is constant (more agents add no further benefit, but also no friction).}
    \label{fig:panel4-sync}
\end{figure*}


\begin{figure*}[!htbp]
    \centering
    \includegraphics[width=\textwidth]{validation/figures/panel_5.png}
    \caption{\textbf{Partition Extinction and the Congruence of Phase-Locked Trajectories.}
    (\textbf{A})~Categorical filtering geometries: angular field magnitude in the $(x,y)$ plane for the multipole orders $\ell=0$ (isotropic, steelblue), $\ell=1$ (two lobes, seagreen), and $\ell=2$ (four lobes, darkorange), the angular index of the partition coordinates $(n,\ell,m,s)$ of \Cref{def:partition-coords}. These illustrate the categorical observables of \Cref{def:observables} on which distinguishability is defined.
    (\textbf{B})~Pairwise categorical-indistinguishability matrix for a $5$-agent phase-locked ensemble ($\phi_i=\phi_j$ for all pairs): green entries mark pairs that are categorically indistinguishable. The matrix is uniformly green, the content of \Cref{prop:correspondence}: in a phase-locked ensemble every pair is categorically indistinguishable and the inter-agent partition lag vanishes (\Cref{thm:extinction}).
    (\textbf{C})~Trajectory traces $M(t)=\int_0^t g(t')\,\dd t'$ for four phase-locked agents driven by a common field with small ($\sigma=0.02$) per-agent noise; the four trajectories are nearly indistinguishable within a tolerance set by the noise. This illustrates the structural correspondence of \Cref{prop:correspondence}: congruence of coordinated trajectories is a \emph{consequence} of synchronisation, and---by the observable-commutation theorem (\Cref{thm:commutation})---is functional (categorical) equivalence, not loss of physical identity.
    (\textbf{D})~Three-dimensional synchronisation diagram: stable order parameter $R^*=\sqrt{1-K_c/K}$ over $(K,\sigma_\omega)\in[0.1,4]\times[0.1,2.0]$, $25\times 25$ grid, viridis colourmap, with $K_c=2/[\pi g(0)]$ of \Cref{thm:Kc}. The surface rises with coupling $K$ and falls with frequency spread $\sigma_\omega$; it is zero on the critical-coupling locus $K=K_c$, above which synchronisation is possible.}
    \label{fig:panel5-extinction}
\end{figure*}
